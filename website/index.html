module data_processor

// `var` 表示对参数的可变借用，不会发生拷贝，且不可逃逸
fn process_inplace(data: var int32[]) =
    if len(data) > 0:
        data[0] = data[0] * 2

fn main(): int32 =
    // 默认 move 语义分配，基于 Arena 不用 GC
    let numbers = [10, 20, 30]
    
    // 安全入参：自动转为限制借用
    process_inplace(numbers)
    
    if numbers[0] == 20:
        return 0
    return 1
import std/threading

fn worker(counter: Arc[Mutex[int32]]) =
    // Mutex 获取借用权
    let locked: var int32 = counter.lock()
    locked = locked + 1

fn main(): int32 =
    // 跨线程共享必须通过受限模型 Arc 包装
    let shared_counter = Arc.new(Mutex.new(0))
    
    let t1 = threading.spawn(worker, shared_counter)
    let t2 = threading.spawn(worker, shared_counter)
    
    t1.join()
    t2.join()
    return 0

        let user = user_opt.unwrap()
        if user.id == 42:
            return 0
    return 1
