import cheng.arch.riscv
import cheng.rtos.freertos

// Hardware Interrupt Handler (RISC-V Machine Mode)
@interrupt(source=RISCV_M_TIMER)
fn handle_timer_irq() {
    // Zero-latency context switch
    Scheduler.tick()
}

@entry
fn main() {
    // Bare-metal RISC-V startup
    System.init_clocks()
    
    // Create Real-Time Task
    @task(priority=HIGH, stack=1024)
    fn telemetry_task() {
        loop {
            // Read sensor via raw pointers (MMIO)
            let raw_val = volatile_read(0x4000_1000)
            Gpio.toggle(LED_PIN)
            Task.delay(100ms)
        }
    }
    
    Scheduler.start()
}
